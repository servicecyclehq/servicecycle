# ServiceCycle Python extraction dependencies
# Pin versions to prevent silent regressions on Docker rebuild.
# To update: test with new version, update this file, rebuild Docker image.
#
# All packages must be available as pre-built wheels (--only-binary=:all:)
# for the alpine linux/amd64 target. If a new version drops wheel support,
# the Docker build will fail loudly rather than silently compiling from source.

pdfplumber==0.11.4
pypdf==4.3.1
pytesseract==0.3.13
Pillow>=10.4.0,<12.0
# pypdfium2 is intentionally omitted: it ships only manylinux (glibc) wheels and
# cannot be imported on Alpine musl. pdfplumber will pull it in as a transitive
# dep (pip install succeeds because the wheel downloads fine), but the Dockerfile
# wraps the import check in a fail-open block so a pypdfium2 ImportError on musl
# is logged as a warning rather than failing the build. Machine-readable PDF
# text extraction (pdfplumber + pdfminer.six) still works; scanned-PDF OCR falls
# back to the pdfjs parser. To restore full OCR, switch to node:20-slim (Debian).
